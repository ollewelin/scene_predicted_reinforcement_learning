% Read data from the log file
fid = fopen("output_win.txt", "r");
data = textscan(fid, "%f %f %f %f", "headerlines", 1);
fclose(fid);
%epoch Now_win_probablilty Old_win_probablilty total_plays
epoch = data{1};  % Extract the first column (epoch)
Now_win_probablilty = data{2};  % Extract the second column 
Old_win_probablilty = data{3};  % Extract the third column

% Create a new figure
figure;

plot(epoch, Now_win_probablilty, 'g.-', 'LineWidth', 1.5);  % Green line
hold on;
plot(epoch, Old_win_probablilty, 'r.-', 'LineWidth', 1.5);  % Red line

% Add labels and title
xlabel("Epoch");
ylabel("Values");
title("Data Trends vs. Epoch");

% Show grid
grid on;

% Legend
legend("Now_win_probablilty", "Last Win Probability");

% Release hold
hold off;


% Legend
legend([h1, h2], "Now_win_probablilty", "Last Win Probability");

% Release hold
hold off;
